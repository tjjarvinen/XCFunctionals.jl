Tutorial

List all functionals

using XCFunctionals

list_functionals()
gga_example
lda_example

Search for functionals with specific name

list_functionals("gga")
gga_example

Generate symbolic represenations for functionals

xc = generate_functional("lda_example")
xc.xc

\begin{equation} \left( \rho - b \right) e^{ - \left( \rho - b \right)^{2} a} \end{equation}

Derivatives

generate_functional_derivatives!(xc, 2)
println(xc.derivatives[1])
println(xc.derivatives[2])
exp(-a*((ρ - b)^2)) - a*(ρ - b)*(2ρ - 2b)*exp(-a*((ρ - b)^2))
(a^2)*(ρ - b)*((2ρ - 2b)^2)*exp(-a*((ρ - b)^2)) - 2a*(ρ - b)*exp(-a*((ρ - b)^2)) - 2a*(2ρ - 2b)*exp(-a*((ρ - b)^2))

Generate C code

xc = build_functional("lda_example", 1,2,3; target=CTarget())

# First derivative
print(xc[1])
#include <math.h>
void diffeqf(double* du, const double RHS1) {
  du[0] = -2 * (RHS1 * RHS1) * exp(-1 * (RHS1 * RHS1)) + exp(-1 * (RHS1 * RHS1));
}

Compile Julia code

xc = build_functional("lda_example", 1,2,3; compile=true)

# First derivative

GPU calculation

Standard broadcasting works with GPUs

using Cuda
ρ = rand(3)
ρ_cuda = CuArray(ρ)
xc[1].( ρ_cuda )
using AMDGPU
ρ = rand(3)
ρ_rocm = ROCArray(ρ)
xc[1].( ρ_rocm )
using OneAPI
ρ = rand(3)
ρ_one = oneArray(ρ)
xc[1].( ρ_one )
